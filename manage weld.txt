Option Explicit '初始化对象
Dim CATIA As Object
Dim xlApp As Object
Dim oDoc
Sub intersel()
	Set CATIA = GetObject(, "CATIA.Application") '获取catia程序
	Set xlApp = GetObject(, "Excel.Application") '获取excel程序
	Set oDoc = CATIA.ActiveDocument
	Dim iPrd, rootPrd, oPrd	
'========获取点焊部件===========
	dim wdpn,wprd: wdpn="12000"
	Dim bDict, children, i, pn
	'Set oPrd = products1.AddNewComponent("Product", wdpn)  '插入部件
	' Set oprd = products1.AddNewProduct("") ‘插入产品
	Set children = rootPrd.Products 
	Set bDict = CreateObject("Scripting.Dictionary") 形成部件字典
		For i = 1 To children.Count
			pn = children.Item(i).PartNumber
			If bDict.exists(pn) = False Then
				bDict(pn) = children.Item(i).ReferenceProduct
			End If
		next
	If bDict.exists(wdpn) = true then '获取焊点部件
		set wPrd=bDict(wdpn)  
	Else
		Set wPrd = rootPrd.Products.AddNewComponent("Product", wdpn)  '插入焊接信息部件'	
	end if  '获取焊点部件
'========获取点焊部件===========
'========选取和生成===========
	Dim oSel, Status
	Dim i, iType(0), oPn, iPn
	Set oSel = oDoc.Selection
	oSel.Clear
	iType(0) = "Product"
    Status = oSel.SelectElement3(iType, "选择被连接产品", True, 2, False)
    If Status = "Cancel" Then
        MsgBox "操作取消"
        Exit Sub
    End If
    If Status = "Normal" And oSel.Count2 <= 3 Then
        For i = 1 To oSel.Count
            oPn = oPn & "_" & oSel.Item(i).LeafProduct.PartNumber
        Next
         iPn = "SotWeld_" & oPn
		Set oPrd = wPrd.Products.AddNewComponent("Part", "ipn")
            MsgBox "已创建" & iPn
		oSel.Clear	
    End If
end sub
